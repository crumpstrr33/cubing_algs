% Name and Alg for some move with optional pic
\NewDocumentCommand{\nameAndAlg}{ O{} O{} m m }{
	\item[--]\textbf{\textcolor{algName}{#3}:}
	\ifthenelse{\isempty{#1}}{
		\textcolor{alg}{#4}
	}{
		\begin{minipage}[t]{\linewidth}
			\adjustbox{valign=c}{
				\includegraphics[width=10mm, height=10mm]{#1}
			}
			\textcolor{alg}{#4}
		\end{minipage}
	}
	\ifthenelse{\isempty{#2}}{}{
	
		\info{#2}
	}
}

% Alg name and the regular and OH and algs
\NewDocumentCommand{\nameAndAlgAndOH}{ O{} O{} O{} m m m }{
	\item[--]\textbf{\textcolor{algName}{#4}: }
	\ifthenelse{\isempty{#1}}{
		% If no picture
		\textcolor{alg}{#5}
		
		% Two-handed info
		\ifthenelse{\isempty{#2}}{\info{ }}{\info{#2}}

		\begin{footnotesize}
			\textcolor{algNameOH}{\textbf{OH}}: \textcolor{algOH}{#6}
		\end{footnotesize}

		% One-handed info
		\ifthenelse{\isempty{#3}}{\info{ }}{\info{#3}}
	}{
		% If there is a picture
		\begin{tabular}{p{1.6cm} p{12cm}}
			\vspace{0pt}
			\includegraphics[width=18mm, height=18mm]{#1}
			&
			\vspace{0pt}
			\textcolor{alg}{#5}

			% Two-handed info
			\ifthenelse{\isempty{#2}}{\info{ }}{\info{#2}}

			\begin{footnotesize}
				\textcolor{algNameOH}{\textbf{OH}}: \textcolor{algOH}{#6}
			\end{footnotesize}

			% One-handed info
			\ifthenelse{\isempty{#3}}{\info{ }}{\info{#3}}
		\end{tabular}
	}
}

\subsection{PBL}
\begin{quote}
On the top and bottom face when at this step, 2 corners will be solved (with some AUF) or all 4 will be. "Edges" is when the 2 that are solved are adjacent to each other. "Corners" is when the 2 that are solved are not next to each other and so are on opposite corners.
\end{quote}
\begin{itemizeSquish}
	\nameAndAlg{Corners Up, Solved Down}{Y-perm (see 3x3 1L PLL)}
	\nameAndAlg{Edges Up, Solved Down}{T-perm (see 3x3 1L PLL)}
	\nameAndAlg[][Edges in front]{Edges Up, Edges Down}{R2 U' B2 U2 R2 U' R2}
	\nameAndAlg{Corners Up, Corners Down}{R2 B2 R2}
	\nameAndAlg[][Edges on left or right]{Corners Up, Edges Down}{(R2 U' R2 U)x2 R2}
\end{itemizeSquish}

\section{Megaminx}
\subsection{Beginner's Method}
This method requires no algorithms until the last layer where it closely resembles the 3x3 beginner's method due to it being just simple communtators for the most part.
\begin{itemizeSquish}
	\nameAndAlg{Orient Edges}{F R U R' U' F'}
	\nameAndAlg[][Is the 3x3 Sune equivalent. Counterclockwise rotates BL, BR and FR edge pieces]{Permutate Edge}{R U R' U R U2' R'}
	\nameAndAlg[][Repeat communtator until corner is solve then AUF to next unsolved corner]{Orient Corners}{R' DR' R DR}
	\nameAndAlg[][Remove corner with the first algorithm, AUF to its correct slot and insert corner with the second algorithm]{Permutate Corners}{R' DR' R and R' DR R}
\end{itemizeSquish}
